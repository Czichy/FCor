Numerical Computing with F# and MKL

Adam Mlocek, StatFactory Ltd UK

@FSharpNumerics

About me

  • SAS dev and stat modelling in insurance/reinsurance (before 2005)
  • C# developer since 2002
  • F# developer since 2010
  • Founder of StatFactory Ltd
  • Co-creator of FCell: .NET integration with Excel
  • Master's degree in Applied Math

I Sho it before

  • R
  • Matlab
  • SAS
  • numpy
  • NMath
  • Math.NET

I actually wanted this:

  • expressive and concise, I don't like typing
  • feeling natural for math and stat people
  • but strongly and statically typed, I am often dislexic
  • but with minimum type annotations (did I say concise?)
  • I paid for RAM, I want access to all of it
  • it's 2015, my data could be int64, not int16

F# DSL for math and stat: codename 'FCor'

  • open source (MIT)
  • C/MKL allocated memory, int64 size
  • Intel MKL: BLAS, LAPACK, RNG...
  • free Community version of MKL
  • heavily overloaded F# operators and functions
  • packaged as 1 .NET dll, available from NuGet
  • 1.0 released, 1.0.1 beta
  • source code: FCor on GitHub
  • to be renamed to 'ξ' (Xi: ksi)

function value = binomAmerican(S0, K, r, vol, T, n)
    dt = T / n;
    u = exp(vol*sqrt(dt));
    pUp = (u*exp(r*dt) - 1.0) / (u*u - 1.0) * exp(-r*dt);
    pDown = exp(-r*dt) - pUp;
    u2 = (1.0/(u*u))*ones(n+1,1);
    u2(1,:) = 1.0;
    S = S0 *(u^n)*cumprod(u2, 1);
    K2 = repmat(K, n+1,1);
    V = K2 - S;
    V(V < 0) = 0;
    for i = n:-1:1
        V1 = V(1:i);
        V2 = V(2:i+1);
        V(1:i) = pUp * V1 + pDown * V2;
        S = S / u;
        V = max(V, K2-S);
    end  
    value = V(1);
end

let binomAmerican (S0 : float) (K : float) (r : float) (vol : float) (T : float) n =
    let dt = T / float(n)
    let u = exp(vol * sqrt(dt))
    let pUp = (u*exp(r*dt) - 1.0) / (u*u - 1.0) * exp(-r*dt)
    let pDown = exp(-r*dt) - pUp
    use u2 = new Vector(n+1, 1.0/(u*u))
    u2.[0L] <- 1.0
    use S = S0 *(u**float(n)) * (cumprod u2)
    use K2 = new Vector(n+1, K)
    use V = K2 - S
    evalIn (iif (V.AsExpr .< 0.0) !!0.0 V) V
    let expr1 = S.AsExpr / u
    let expr2 = VectorExpr.Max(V.AsExpr, (K2-S.AsExpr))
    for i in n..(-1)..1 do
        use V1 = V.View(0,i-1)
        use V2 = V.View(1,i)
        evalIn (pUp * V1.AsExpr + pDown * V2.AsExpr) V1
        evalIn expr1 S
        evalIn expr2 V
    V.[0]
